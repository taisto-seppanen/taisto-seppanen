    export async function setNewFilm(newfilm) {
      await firebase
        .database()
        .ref("films/")
        .set({ newfilm });
    }

    export async function getFilms() {
      let filmsArr = [];
      await firebase
        .database()
        .ref("films/")
        .get()
        .then(snapshot => {
          if (snapshot.exists()) {
            filmsArr = snapshot.val().newfilm;
          } else {
            console.warn("bad request");
          }
        });
      return filmsArr;
    }
